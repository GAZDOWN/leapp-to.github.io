Schema Definitions

TargetHostBase: object

Base schema for target_host field

target_host: string localhost
Example
{
  "target_host": "string"
}

SourceHostBase: object

Base schema for source_host field

source_host: string
Example
{
  "source_host": "string"
}

PortListItemBase: object

A map of ports with their respective data (can be just an empty map)

port: string
port_data: object
Example
{
  "port": "string",
  "port_data": {
    "name": "string"
  }
}

PortListBase: object

A map of TCP/UDP ports

tcp: PortListItemBase
udp: PortListItemBase
Example
{
  "tcp": {
    "port": "string",
    "port_data": {
      "name": "string"
    }
  },
  "udp": {
    "port": "string",
    "port_data": {
      "name": "string"
    }
  }
}

PortMappingItemBase: object

protocol: string tcp, udp

Port protocol

exposed_port: integer

Port on which the service will be mapped to

port: integer

Port on which the service is currently listening

Example
{
  "protocol": "string",
  "exposed_port": "integer",
  "port": "integer"
}

PortMappingBase: object

ports: PortMappingItemBase
Example
{
  "ports": [
    {
      "protocol": "string",
      "exposed_port": "integer",
      "port": "integer"
    }
  ]
}

DestroyContainerInput: object

Destroy/Delete container request

container_name: string
target_user: string root
    TargetHostBase
Example
{
  "container_name": "string",
  "target_user": "string"
}

PortInspectInput: object

Inspect ports request

scan_options: object
    TargetHostBase
Example
{
  "scan_options": {
    "port_range": "string",
    "shallow_scan": "boolean"
  }
}

PortMapInput: object

Inspect ports request

tcp_ports: PortMappingBase
excluded_tcp_ports: PortListBase
default_port_map: boolean true
    TargetHostBase SourceHostBase
Example
{
  "tcp_ports": {
    "ports": [
      {
        "protocol": "string",
        "exposed_port": "integer",
        "port": "integer"
      }
    ]
  },
  "excluded_tcp_ports": {
    "tcp": {
      "port": "string",
      "port_data": {
        "name": "string"
      }
    },
    "udp": {
      "port": "string",
      "port_data": {
        "name": "string"
      }
    }
  },
  "default_port_map": "boolean"
}

CheckTargetInput: object

Check target request

target_user_name: string root
check_target_service_status: boolean
    TargetHostBase
Example
{
  "target_user_name": "string",
  "check_target_service_status": "boolean"
}

Result: object

Default API response

errors: Error
data: object
Example
{
  "errors": [
    {
      "code": "integer",
      "message": "string"
    }
  ],
  "data": "object"
}

Error: object

code: integer
message: string
Example
{
  "code": "integer",
  "message": "string"
}
